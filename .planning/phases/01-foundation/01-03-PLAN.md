1. Refine `packages/plugin-effect/src/resolver.ts` error handling.

The current `createResolver()` from Plan 01-02 throws `exit.cause` directly on
failure. Improve this so the error message is descriptive and works with Fresh's
error mechanisms.

First, update the `ResolverOptions` interface JSDoc to document that `mapError`
receives a `Cause<E>` from Effect, not a plain error:

```typescript
export interface ResolverOptions {
  /**
   * Custom error mapper for Effect failures. Receives `exit.cause` which is
   * an Effect `Cause<E>` value — a structured wrapper around the failure, NOT
   * a plain Error. Use `Cause.squash(cause)` or pattern-match on the Cause
   * variants (Fail, Die, Interrupt) to extract the underlying error.
   *
   * If omitted, the resolver throws a standard Error with the Cause preserved
   * in `error.cause`, which Fresh routes to its error page (_error.tsx) or
   * DEFAULT_ERROR_HANDLER (returns 500).
   */
  mapError?: (cause: unknown) => Response;
}
```

Then update the failure path in `createResolver()` to extract a readable message
from the Effect Cause before re-throwing. Replace the existing failure section
(after `if (Exit.isSuccess(exit))`) with:

```typescript
// Failure path
if (options.mapError) {
  return options.mapError(exit.cause);
}

// Default: extract a readable error from the Cause and throw it.
// This propagates through Fresh's middleware chain:
// - segmentMiddleware catches it and renders _error.tsx if present
// - Otherwise DEFAULT_ERROR_HANDLER in app.ts returns 500
//
// Note: exit.cause is a Cause<E> — a structured Effect wrapper, not a plain Error.
// We wrap it in a standard Error so Fresh's error handling works correctly.
const cause = exit.cause;
let message = "Effect handler failure";

// Cause.pretty() or Cause.squash() may exist in v4 — try squash first
// to get a plain Error, then fall back to string coercion.
if (typeof cause === "object" && cause !== null) {
  if ("message" in cause && typeof (cause as any).message === "string") {
    message = (cause as any).message;
  } else {
    try {
      message = String(cause);
    } catch {
      // keep default message
    }
  }
}

const error = new Error(message);
error.cause = cause;
throw error;
```

This ensures:

- Fresh's `DEFAULT_ERROR_HANDLER` receives a standard Error (logs via
  console.error)
- Fresh's `segmentMiddleware` catch block fires (renders _error.tsx)
- The original Effect Cause is preserved in `error.cause` for debugging
- The thrown error is NOT an HttpError (status defaults to 500 via the catch
  block)

// Integration test: Effect handlers through Fresh's full request path.
// Uses the same App + FakeServer pattern as Fresh's own app_test.tsx.

// SC-1: Effect-returning handler produces same HTTP response as async handler
Deno.test("Integration: Effect.succeed(Response) produces same response as async handler", async () => {
  const app = new App()
    .use(effectPlugin())
    .get("/async", () => new Response("hello from async"))
    .get("/effect", () => Effect.succeed(new Response("hello from effect")));

  const server = new FakeServer(app.handler());

  const asyncRes = await server.get("/async");
  const effectRes = await server.get("/effect");

  // Same status
  expect(asyncRes.status).toEqual(200);
  expect(effectRes.status).toEqual(200);

  // Same body content
  const asyncBody = await asyncRes.text();
  const effectBody = await effectRes.text();
  expect(asyncBody).toEqual("hello from async");
  expect(effectBody).toEqual("hello from effect");
});

// SC-1 variant: Effect handler returning Response with custom status/headers
Deno.test("Integration: Effect handler with custom status and headers", async () => {
  const app = new App()
    .use(effectPlugin())
    .get("/", () =>
      Effect.succeed(
        new Response("created", {
          status: 201,
          headers: { "X-Custom": "test" },
        }),
      ));

  const server = new FakeServer(app.handler());
  const res = await server.get("/");

  expect(res.status).toEqual(201);
  expect(await res.text()).toEqual("created");
  expect(res.headers.get("X-Custom")).toEqual("test");
});

// SC-3: Unhandled Effect failure renders Fresh error page (500), not crash
Deno.test("Integration: Effect.fail produces 500 response, not unhandled rejection", async () => {
  const app = new App()
    .use(effectPlugin())
    .get("/", () => Effect.fail("something went wrong"));

  const server = new FakeServer(app.handler());
  const res = await server.get("/");

  // Fresh's DEFAULT_ERROR_HANDLER catches the thrown Error and returns 500
  expect(res.status).toEqual(500);
  const body = await res.text();
  expect(body).toContain("Internal server error");
});

// SC-3 variant: Effect.die (defect) also produces 500, not crash
Deno.test("Integration: Effect.die produces 500 response, not unhandled rejection", async () => {

    .use(effectPlugin())
    .get("/", () => Effect.die(new Error("unexpected defect")));

  const server = new FakeServer(app.handler());
  const res = await server.get("/");

  expect(res.status).toEqual(500);

  expect(body).toContain("Internal server error");
