1. Refine `packages/plugin-effect/src/resolver.ts` error handling.

   The current `createResolver()` from Plan 01-02 throws `exit.cause` directly on failure.
   Improve this so the error message is descriptive and works with Fresh's error mechanisms:

   Update the failure path in `createResolver()` to extract a readable message from
   the Effect Cause before re-throwing. Replace the existing failure section (after
   `if (Exit.isSuccess(exit))`) with:

   ```typescript
   // Failure path
   if (options.mapError) {
     return options.mapError(exit.cause);
   }

   // Default: extract a readable error from the Cause and throw it.
   // This propagates through Fresh's middleware chain:
   // - segmentMiddleware catches it and renders _error.tsx if present
   // - Otherwise DEFAULT_ERROR_HANDLER in app.ts returns 500
   const cause = exit.cause;
   let message = "Effect handler failure";

   // Cause.pretty() or Cause.squash() may exist in v4 — try squash first
   // to get a plain Error, then fall back to string coercion.
   if (typeof cause === "object" && cause !== null) {
     if ("message" in cause && typeof (cause as any).message === "string") {
       message = (cause as any).message;
     } else {
       try {
         message = String(cause);
       } catch {
         // keep default message
       }
     }
   }

   const error = new Error(message);
   error.cause = cause;
   throw error;
   ```

   This ensures:
   - Fresh's `DEFAULT_ERROR_HANDLER` receives a standard Error (logs via console.error)
   - Fresh's `segmentMiddleware` catch block fires (renders _error.tsx)
   - The original Effect Cause is preserved in `error.cause` for debugging
   - The thrown error is NOT an HttpError (status defaults to 500 via the catch block)

   ```typescript
     assertEquals(isEffect(eff), true);
   });

   Deno.test("isEffect returns false for null", () => {
     assertEquals(isEffect(null), false);
   });

   Deno.test("isEffect returns false for undefined", () => {
     assertEquals(isEffect(undefined), false);

   Deno.test("isEffect returns false for plain objects", () => {
     assertEquals(isEffect({}), false);
     assertEquals(isEffect({ data: "hello" }), false);
   });

   Deno.test("isEffect returns false for Response objects", () => {
     assertEquals(isEffect(new Response("ok")), false);
   });

   Deno.test("isEffect returns false for Promises", () => {
     assertEquals(isEffect(Promise.resolve(42)), false);
   });

     const eff = Effect.fail("boom");
     assertEquals(isEffect(eff), true);

     const runtime = ManagedRuntime.make(Layer.empty);
     try {
       const resolver = createResolver(runtime);
       const response = new Response("hello");
       const result = await resolver(response, {});
       assertEquals(result, response);
     } finally {
       await runtime.dispose();
     }
   });

   Deno.test("resolver passes through PageResponse-like objects unchanged", async () => {
     const runtime = ManagedRuntime.make(Layer.empty);
     try {
       const resolver = createResolver(runtime);

       const result = await resolver(pageResponse, {});
       assertEquals(result, pageResponse);
     } finally {
       await runtime.dispose();
     }

     const runtime = ManagedRuntime.make(Layer.empty);
     try {
       const resolver = createResolver(runtime);
       const response = new Response("from effect");
       const eff = Effect.succeed(response);
       const result = await resolver(eff, {});
       assertEquals(result, response);
     } finally {
       await runtime.dispose();
     }

     const runtime = ManagedRuntime.make(Layer.empty);

       const pageResponse = { data: { count: 42 } };

       const result = await resolver(eff, {});

     } finally {
   ```

```typescript
import { assertEquals, assertExists } from "jsr:@std/assert@1";
import { Effect, Layer, Context as EffectContext } from "effect";
import { effectPlugin } from "../src/mod.ts";

// --- effectPlugin() zero-config ---

Deno.test("effectPlugin() returns a middleware function", () => {
  const middleware = effectPlugin();
  assertEquals(typeof middleware, "function");
});

Deno.test("effectPlugin() zero-config: middleware sets ctx.state.effectRuntime", async () => {
  const middleware = effectPlugin();

  // Create a minimal ctx mock
  const state: Record<string, unknown> = {};
  const ctx = {
    state,
    next: () => Promise.resolve(new Response("ok")),
  };

  const response = await middleware(ctx as any);
  assertExists(state.effectRuntime);
  assertEquals(response instanceof Response, true);
});

// --- effectPlugin({ layer }) with custom Layer ---

// Define a simple test service
class GreetingService extends EffectContext.Tag("GreetingService")<
  GreetingService,
  { readonly greet: (name: string) => string }
>() {}

const TestLayer = Layer.succeed(GreetingService, {
  greet: (name: string) => `Hello, ${name}!`,
});

Deno.test("effectPlugin({ layer }) accepts a typed Layer", () => {
  const middleware = effectPlugin({ layer: TestLayer });
  assertEquals(typeof middleware, "function");
});

Deno.test("effectPlugin({ layer }) middleware sets ctx.state.effectRuntime", async () => {
  const middleware = effectPlugin({ layer: TestLayer });

  const state: Record<string, unknown> = {};
  const ctx = {
    state,
    next: () => Promise.resolve(new Response("ok")),
  };

  await middleware(ctx as any);
  assertExists(state.effectRuntime);
});

// --- effectPlugin mapError option ---

Deno.test("effectPlugin({ mapError }) passes mapError to resolver", () => {
  const mapError = (_cause: unknown) => new Response("error", { status: 500 });
  // This should not throw — just verifying it accepts the option
  const middleware = effectPlugin({ mapError });
  assertEquals(typeof middleware, "function");
});
```
