**Step C: Create `packages/plugin-effect/src/hydration.ts`**

This module exports `setAtom()`, `serializeAtomHydration()`, and
`initAtomHydrationMap()`.

```typescript
import * as Atom from "effect/unstable/reactivity/Atom";

// Symbol used as the key on ctx.state for the per-request hydration map
export const ATOM_HYDRATION_KEY = Symbol.for("fresh_atom_hydration");

/**
 * Set an atom value for server-to-client hydration.
 *
 * The atom MUST be created with Atom.serializable({ key, schema }).
 * The value is stored in ctx.state and serialized into the HTML response.
 * On the client, the island's AtomRegistry picks up the pre-seeded value
 * automatically via registry.setSerializable().
 *
 * @throws Error if atom is not serializable (missing Atom.serializable wrapper)
 * @throws Error if atom key is already set within the same request
 * @throws Error if hydration map not initialized (effectPlugin middleware missing)
 */
export function setAtom<A>(
  ctx: { state: unknown },
  atom: Atom.Atom<A>,
  value: A,
): void {
  // Validate: atom must be serializable
  if (!Atom.isSerializable(atom)) {
    throw new Error(
      "setAtom() requires a serializable atom. " +
        "Wrap with Atom.serializable({ key, schema }).",
    );
  }

  // Access the Serializable metadata to get key and encode function.
  // Atom.isSerializable confirms SerializableTypeId exists on the atom.
  // deno-lint-ignore no-explicit-any
  const serializable = (atom as any)[Atom.SerializableTypeId] as {
    readonly key: string;
    readonly encode: (value: A) => unknown;
  };

  const key = serializable.key;

  // Get per-request map
  const state = ctx.state as Record<string | symbol, unknown>;
  const map = state[ATOM_HYDRATION_KEY] as Map<string, unknown> | undefined;
  if (!map) {
    throw new Error(
      "setAtom() called but atomHydration map not initialized. " +
        "Is effectPlugin() middleware active?",
    );
  }

  // Duplicate key detection (per-request)
  if (map.has(key)) {
    throw new Error(
      `Duplicate atom key "${key}" in the same request. ` +
        "Each atom must have a unique key.",
    );
  }

  // Encode using the atom's Schema and store
  const encoded = serializable.encode(value);
  map.set(key, encoded);
}

/**
 * Serialize the per-request atom hydration map into a JSON string.
 * Returns null if no atoms were set.
 */
export function serializeAtomHydration(
  ctx: { state: unknown },
): string | null {
  const state = ctx.state as Record<string | symbol, unknown>;
  const map = state[ATOM_HYDRATION_KEY] as Map<string, unknown> | undefined;

  if (!map || map.size === 0) return null;

  const obj: Record<string, unknown> = {};
  for (const [key, encoded] of map) {
    obj[key] = encoded;
  }

  return JSON.stringify(obj);
}

/**
 * Initialize the per-request atom hydration map on ctx.state.
 * Called by effectPlugin middleware at the start of each request.
 */
export function initAtomHydrationMap(ctx: { state: unknown }): void {
  (ctx.state as Record<string | symbol, unknown>)[ATOM_HYDRATION_KEY] = new Map<
    string,
    unknown
  >();
}
```

Note on Atom.SerializableTypeId: Verify the actual export from
`effect/unstable/reactivity/Atom` at v4.0.0-beta.0. If `SerializableTypeId` is
not a direct export, check for alternative access patterns. The research
confirmed `Atom.isSerializable(atom)` and `atom[Atom.SerializableTypeId]` as the
v4 API. Adapt if the actual .d.ts differs.
