const Api = HttpApi.make("testApi").add(

**Step 2: Create `packages/examples/effect-integration/services/api.ts`**

This file defines a small HttpApi that wraps the existing `TodoService` -- demonstrating
how HttpApi integrates with the app's existing service layer.

```typescript
/**
 * HttpApi definition for the Todo API.
 *
 * Demonstrates:
 * - Schema-first endpoint declarations with typed params and success/error schemas
 * - HttpApiBuilder.group for type-safe handler implementations
 * - HttpApiError.NotFound for typed 404 responses
 */

import { Effect, Schema } from "effect";
import {
  HttpApi,
  HttpApiBuilder,
  HttpApiEndpoint,
  HttpApiError,
  HttpApiGroup,
} from "effect/unstable/httpapi";
import { TodoService } from "./TodoService.ts";

// ---- Schemas ----

const TodoSchema = Schema.Struct({
  id: Schema.String,
  text: Schema.String,
  done: Schema.Boolean,
});

// ---- API Definition ----

export const TodoApi = HttpApi.make("todoApi").add(
  HttpApiGroup.make("todos")
    .prefix("/todos")
    .add(
      HttpApiEndpoint.get("list", "/", {
        success: Schema.Array(TodoSchema),
      })
    )
    .add(
      HttpApiEndpoint.get("getById", "/:id", {
        params: Schema.Struct({ id: Schema.String }),
        success: TodoSchema,
        error: HttpApiError.NotFound,
      })
    )
    .add(
      HttpApiEndpoint.post("create", "/", {
        payload: Schema.Struct({ text: Schema.String }),
        success: TodoSchema,
      })
    )
);

// ---- Group Implementation ----

export const TodosLive = HttpApiBuilder.group(TodoApi, "todos", (handlers) =>
  handlers
    .handle("list", () =>
      Effect.gen(function* () {
        const svc = yield* TodoService;
        return yield* svc.list();
      })
    )
    .handle("getById", ({ params }) =>
      Effect.gen(function* () {
        const svc = yield* TodoService;
        const todos = yield* svc.list();
        const todo = todos.find((t) => t.id === params.id);
        if (!todo) return yield* new HttpApiError.NotFound({});
        return todo;
      })
    )
    .handle("create", ({ payload }) =>
      Effect.gen(function* () {
        const svc = yield* TodoService;
        return yield* svc.create(payload.text);
      })
    )
);
```

**Step 3: Update `packages/examples/effect-integration/main.ts`**

Add the HttpApi mounting. The existing code creates an EffectApp with `AppLayer` (which
provides `TodoService`). Add `httpApi()` call BEFORE `.use(staticFiles())`.

IMPORTANT: `TodosLive` requires `TodoService`, which is provided by `AppLayer`. These layers
must be composed with `Layer.provide` (NOT `Layer.mergeAll`) because `TodosLive` has a
dependency on `AppLayer`'s output. `Layer.mergeAll` builds layers concurrently, so
`TodoService` would not be available to `TodosLive` during concurrent construction.

Pre-compose the group layer with its dependencies using `Layer.provide`:

```typescript
import { HttpError, staticFiles } from "@fresh/core";
import { createEffectApp } from "@fresh/effect";
import { Cause, Layer } from "effect";
import { AppLayer } from "./services/layers.ts";
import { NotFoundError } from "./services/errors.ts";
import { TodoApi, TodosLive } from "./services/api.ts";

// Pre-compose group layer with its dependencies for correct build order.
// Layer.provide ensures TodosLive receives TodoService from AppLayer.
// Do NOT use Layer.mergeAll here — it builds concurrently, so TodosLive
// would fail because TodoService isn't available yet during concurrent build.
const TodosWithDeps = Layer.provide(TodosLive, AppLayer);

export const app = createEffectApp({
  layer: AppLayer,
  mapError: (cause) => {
    const defect = Cause.squash(cause as never);
    if (defect instanceof NotFoundError) {
      throw new HttpError(404);
    }
    throw new HttpError(500);
  },
})
  .httpApi("/api", TodoApi, TodosWithDeps)
  .use(staticFiles())
  .fsRoutes();
```

The HttpApi is mounted at `/api`, so the Todo endpoints are available at:
- `GET /api/todos/` -- list all todos
- `GET /api/todos/:id` -- get todo by id (404 if not found)
- `POST /api/todos/` -- create a new todo

**Why `Layer.provide` and not `Layer.mergeAll`:**

The `httpApi()` method internally does `Layer.mergeAll(...groupLayers)`. If we passed
`TodosLive` and `AppLayer` as separate arguments (`.httpApi("/api", TodoApi, TodosLive, AppLayer)`),
`mergeAll` would attempt to build them concurrently. Since `TodosLive` depends on `TodoService`
from `AppLayer`, concurrent building would fail -- `TodoService` wouldn't be available yet.

By pre-composing with `Layer.provide(TodosLive, AppLayer)`, the resulting `TodosWithDeps` layer
is self-contained: it builds `AppLayer` first (providing `TodoService`), then builds `TodosLive`.
When passed as a single argument to `httpApi()`, `mergeAll` receives just one layer, so
concurrency is not an issue.

The `memoMap` sharing (from Plan 01) ensures that `AppLayer` services built here reuse the
same instances as those built by the main runtime, avoiding duplicate service construction.
