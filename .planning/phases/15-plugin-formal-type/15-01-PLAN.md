1. Create `packages/fresh/src/plugin.ts` with:

```typescript
import type { App } from "./app.ts";

/**
 * A typed plugin that provides routes (as App<S>), documents Effect service
 * requirements (R), and declares the host state shape it requires (S).
 *
 * R is a phantom type parameter — it exists only in the type system to
 * document what Effect services the plugin's handlers require. @fresh/core
 * never imports Effect, so R is unconstrained.
 */
export interface Plugin<Config = unknown, S = unknown, R = never> {
  readonly config: Config;
  readonly app: App<S>;
  /** Phantom type — R appears in the type system only, never at runtime. */
  readonly _phantom?: R;
}

/**
 * Create a typed plugin from a configuration object and a factory function
 * that builds an App<S>.
 *
 * @param config Plugin configuration passed to the factory
 * @param factory Function that receives config and returns an App<S>
 * @returns A Plugin<Config, S, R> that can be mounted via host.mountApp()
 */
export function createPlugin<Config, S, R = never>(
  config: Config,
  factory: (config: Config) => App<S>,
): Plugin<Config, S, R> {
  return { config, app: factory(config) };
}
```

NOTE: The return type annotation on `createPlugin` is explicit (`Plugin<Config, S, R>`) — required for JSR slow-types compliance.

   IMPORTANT: The discrimination check uses `"app" in appOrPlugin && "config" in appOrPlugin && !("handler" in appOrPlugin)` rather than just `"app" in appOrPlugin`. The reason: `App<State>` also has a `config` property, but `App<State>` has a `handler` method while `Plugin` does not. `Plugin` has both `app` and `config` but no `handler`. This three-way check cleanly distinguishes Plugin from App.

   After the `inner` assignment, the rest of the method body stays EXACTLY the same but uses `inner` instead of `app` as the parameter name. Specifically, replace every occurrence of `app.` with `inner.` in the existing method body. The complete replacement:

   ```typescript
   mountApp<Config, R>(path: string, plugin: Plugin<Config, State, R>): this;
   mountApp(path: string, app: App<State>): this;
   mountApp(path: string, appOrPlugin: App<State> | Plugin<unknown, State, unknown>): this {
     const inner: App<State> = "app" in appOrPlugin && "config" in appOrPlugin && !("handler" in appOrPlugin)
       ? (appOrPlugin as Plugin<unknown, State, unknown>).app
       : appOrPlugin as App<State>;

     for (let i = 0; i < inner.#commands.length; i++) {
       const cmd = inner.#commands[i];

       if (cmd.type !== CommandType.App && cmd.type !== CommandType.NotFound) {
         let effectivePattern = cmd.pattern;
         if (inner.config.basePath) {
           effectivePattern = mergePath(inner.config.basePath, cmd.pattern, false);
         }

         const clone = {
           ...cmd,
           pattern: mergePath(path, effectivePattern, true),
           includeLastSegment: cmd.pattern === "/" || cmd.includeLastSegment,
         };
         this.#commands.push(clone);
         continue;
       }

       this.#commands.push(cmd);
     }

     // deno-lint-ignore no-this-alias
     const self = this;
     inner.#getBuildCache = () => self.#getBuildCache();

     for (const reg of inner.#islandRegistrations) {
       this.#islandRegistrations.push(reg);
     }

     if (!this.#effectRunner && inner.#effectRunner) {
       this.#effectRunner = inner.#effectRunner;
     }
     if (!this.#atomHydrationHook && inner.#atomHydrationHook) {
       this.#atomHydrationHook = inner.#atomHydrationHook;
     }

     return this;
   }
   ```

   ```typescript
    * - Plugin.app is an App<S> instance
   ```
