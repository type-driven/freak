    * Either a single Effect handler function or a method map of Effect handlers.
    */
   export type EffectRouteHandler<Data, State, R> =
     | EffectHandlerFn<Data, State, R>
     | EffectHandlerByMethod<Data, State, R>;

   /**
    * The object returned by {@link createEffectDefine}. Provides a `handlers()`
    * method that constrains handler return types to `Effect<..., R>`.
    */
   export interface EffectDefine<State, R> {
     /**
      * Define route handlers constrained to services in R.
      *
      * This function returns the passed input as-is (identity function).
      * The value is in the type constraint: handlers must return
      * `Effect<Response | PageResponse<Data>, unknown, R>`.
      *
      * @typeParam Data - The page data type. Inferred from handlers if not provided.
      * @typeParam Handlers - Always inferred. Do not manually specify.
      */
     handlers<
       Data,
       Handlers extends EffectRouteHandler<Data, State, R> = EffectRouteHandler<
         Data,
         State,
         R
       >,
     >(handlers: Handlers): typeof handlers;
   }
   ```

5. Define options interface:

   ```typescript
   /**
    * Options for {@link createEffectDefine}.
    *
    * @template R - The service requirements type that the Layer provides.
    */
   export interface CreateEffectDefineOptions<R> {
     /**
      * An Effect Layer providing services of type R. When provided,
      * `createEffectDefine` creates its own ManagedRuntime and registers
      * the Effect resolver — making `effectPlugin()` unnecessary.
      *
      * When omitted (type-parameter-only path), no runtime is created.
      * `effectPlugin()` must be registered separately to handle Effect execution.
      *
      * **Important:** The standalone path (`{ layer }`) and `effectPlugin()` are
      * mutually exclusive runtime paths. Using both overwrites the resolver —
      * only the last-registered resolver is active.
      */
     layer?: LayerType.Layer<R, unknown, never>;
   }
   ```

6. Implement the factory:

   ````typescript
   /**
    * Create a typed define object that constrains Effect handler return types
    * to services available in R.
    *
    * This is a compile-time convenience wrapper — the `handlers()` method is
    * an identity function that returns its input unchanged. The value is in
    * the TypeScript type constraints that enforce R at the handler definition site.
    *
    * Two usage paths:
    *
    * **Standalone (with Layer):** Creates own ManagedRuntime, no `effectPlugin()` needed.
    * ```typescript
    * const define = createEffectDefine<AppState, AppServices>({ layer: AppLayer });
    * ```
    *
    * **Type-parameter-only:** Relies on `effectPlugin()` for runtime.
    * ```typescript
    * const define = createEffectDefine<AppState, AppServices>();
    * app.use(effectPlugin({ layer: AppLayer }));
    * ```
    *
    * @typeParam State - The app state type (same as Fresh's createDefine<State>).
    * @typeParam R - The Layer's service requirements type. Defaults to `never` (no services).
    */
   export function createEffectDefine<State = unknown, R = never>(
     options: CreateEffectDefineOptions<R> = {},
   ): EffectDefine<State, R> {
     // Standalone path: create own ManagedRuntime and register resolver
     if (options.layer !== undefined) {
       // deno-lint-ignore no-explicit-any
       const runtime = makeRuntime(
         options.layer as LayerType.Layer<any, any, never>,
       );
       const resolver = createResolver(runtime);
       setEffectResolver(resolver);
       registerDisposal(
         runtime as import("effect").ManagedRuntime.ManagedRuntime<
           unknown,
           unknown
         >,
       );
     }
     // Type-parameter-only path: no runtime setup — effectPlugin() handles it

     return {
       handlers(handlers) {
         return handlers;
       },
     };
   }
   ````

   Note on the `as LayerType.Layer<any, any, never>` cast: This is the same
   pattern used in `effectPlugin()` in `mod.ts` (line 82). The generic `R` is
   erased at the resolver boundary — type safety is enforced at the
   `createEffectDefine` call site, not inside the resolver. Add the
   `deno-lint-ignore no-explicit-any` comment.

   Note on `registerDisposal`: It expects `ManagedRuntime<unknown, unknown>`.
   The runtime from `makeRuntime` is `ManagedRuntime<any, any>` after the Layer
   cast. Cast it to `ManagedRuntime<unknown, unknown>` for the disposal call
   (same pattern as `mod.ts` line 94).

     const define = createEffectDefine<unknown, typeof MsgService>({

     });

     const app = new App()
       .route("/", {
         handler: define.handlers({
           GET: () =>

   IMPORTANT: Use `app.route()` (NOT `app.get()`) — the Effect resolver runs via
   `renderRoute()` in the RouteCommand path only. `app.get()` bypasses
   `renderRoute`.

   IMPORTANT: Pass `define.handlers({ GET: ... }).GET!` to `handler:` —
   `app.route()` expects a single `HandlerFn`, not a method map. The `!`
   non-null assertion is safe because we just defined the GET handler.
